import 'package:flutter/material.dart';
import 'package:flutter/services.dart';
import 'package:shared_preferences/shared_preferences.dart';

void main() => runApp(const AtaApp());

class AtaApp extends StatefulWidget {
  const AtaApp({super.key});
  @override
  State<AtaApp> createState() => _AtaAppState();
}

class _AtaAppState extends State<AtaApp> {
  late AppSettings settings;
  bool _loaded = false;

  @override
  void initState() {
    super.initState();
    _loadSettings();
  }

  Future<void> _loadSettings() async {
    settings = await AppSettings.load();
    setState(() => _loaded = true);
  }

  void _updateSettings(AppSettings newSettings) {
    setState(() => settings = newSettings);
    settings.save();
  }

  @override
  Widget build(BuildContext context) {
    if (!_loaded) {
      return const MaterialApp(
        home: Scaffold(
          backgroundColor: Color(0xFF121212),
          body: Center(child: CircularProgressIndicator()),
        ),
      );
    }

    return MaterialApp(
      debugShowCheckedModeBanner: false,
      title: "ATA Chapters",
      theme: settings.getThemeData(),
      home: AtaChapterPage(settings: settings, onSettingsChanged: _updateSettings),
    );
  }
}

class AppSettings {
  bool isDark;
  String lang;
  String fontFamily;
  double fontSize;
  Color textColor;
  Color backgroundColor;
  Color primaryColor;
  double lineHeight;

  AppSettings({
    this.isDark = true,
    this.lang = 'tr',
    this.fontFamily = 'Roboto',
    this.fontSize = 13.0,
    Color? textColor,
    Color? backgroundColor,
    Color? primaryColor,
    this.lineHeight = 1.3,
  })  : textColor = textColor ?? (isDark ? Colors.white : Colors.black87),
        backgroundColor = backgroundColor ?? (isDark ? const Color(0xFF121212) : const Color(0xFFF5F5F5)),
        primaryColor = primaryColor ?? Colors.blue;

  static Future<AppSettings> load() async {
    final prefs = await SharedPreferences.getInstance();
    final isDark = prefs.getBool('isDark') ?? true;
    return AppSettings(
      isDark: isDark,
      lang: prefs.getString('lang') ?? 'tr',
      fontFamily: prefs.getString('fontFamily') ?? 'Roboto',
      fontSize: prefs.getDouble('fontSize') ?? 13.0,
      textColor: Color(prefs.getInt('textColor') ?? (isDark ? 0xFFFFFFFF : 0xFF212121)),
      backgroundColor: Color(prefs.getInt('backgroundColor') ?? (isDark ? 0xFF121212 : 0xFFF5F5F5)),
      primaryColor: Color(prefs.getInt('primaryColor') ?? 0xFF2196F3),
      lineHeight: prefs.getDouble('lineHeight') ?? 1.3,
    );
  }

  Future<void> save() async {
    final prefs = await SharedPreferences.getInstance();
    await prefs.setBool('isDark', isDark);
    await prefs.setString('lang', lang);
    await prefs.setString('fontFamily', fontFamily);
    await prefs.setDouble('fontSize', fontSize);
    await prefs.setInt('textColor', textColor.value);
    await prefs.setInt('backgroundColor', backgroundColor.value);
    await prefs.setInt('primaryColor', primaryColor.value);
    await prefs.setDouble('lineHeight', lineHeight);
  }

  ThemeData getThemeData() {
    return ThemeData(
      brightness: isDark ? Brightness.dark : Brightness.light,
      scaffoldBackgroundColor: backgroundColor,
      primaryColor: primaryColor,
      colorScheme: ColorScheme.fromSeed(
        seedColor: primaryColor,
        brightness: isDark ? Brightness.dark : Brightness.light,
      ),
      appBarTheme: AppBarTheme(
        backgroundColor: isDark ? const Color(0xFF1E1E1E) : primaryColor,
        foregroundColor: isDark ? Colors.white : Colors.white,
      ),
      cardColor: isDark ? const Color(0xFF1E1E1E) : Colors.white,
      textTheme: TextTheme(
        bodyMedium: TextStyle(
          color: textColor,
          fontFamily: fontFamily,
          fontSize: fontSize,
          height: lineHeight,
        ),
      ),
    );
  }

  AppSettings copyWith({
    bool? isDark,
    String? lang,
    String? fontFamily,
    double? fontSize,
    Color? textColor,
    Color? backgroundColor,
    Color? primaryColor,
    double? lineHeight,
  }) {
    final newIsDark = isDark ?? this.isDark;
    return AppSettings(
      isDark: newIsDark,
      lang: lang ?? this.lang,
      fontFamily: fontFamily ?? this.fontFamily,
      fontSize: fontSize ?? this.fontSize,
      textColor: textColor ?? (isDark != null ? (newIsDark ? Colors.white : Colors.black87) : this.textColor),
      backgroundColor: backgroundColor ?? (isDark != null ? (newIsDark ? const Color(0xFF121212) : const Color(0xFFF5F5F5)) : this.backgroundColor),
      primaryColor: primaryColor ?? this.primaryColor,
      lineHeight: lineHeight ?? this.lineHeight,
    );
  }

  void resetToDefault() {
    isDark = true;
    lang = 'tr';
    fontFamily = 'Roboto';
    fontSize = 13.0;
    textColor = Colors.white;
    backgroundColor = const Color(0xFF121212);
    primaryColor = Colors.blue;
    lineHeight = 1.3;
  }
}

class AtaChapterPage extends StatefulWidget {
  final AppSettings settings;
  final ValueChanged<AppSettings> onSettingsChanged;

  const AtaChapterPage({super.key, required this.settings, required this.onSettingsChanged});

  @override
  State<AtaChapterPage> createState() => _AtaChapterPageState();
}

class _AtaChapterPageState extends State<AtaChapterPage> with TickerProviderStateMixin {
  final TextEditingController _searchCtrl = TextEditingController();
  final Set<String> _favorites = {};
  final Set<String> _expanded = {};
  final ScrollController _scrollController = ScrollController();
  late final AnimationController _listAnimCtrl;
  late final Animation<double> _listFadeAnim;
  bool _favoritesLoaded = false;

  @override
  void initState() {
    super.initState();
    _loadFavorites();
    _listAnimCtrl = AnimationController(vsync: this, duration: const Duration(milliseconds: 600));
    _listFadeAnim = CurvedAnimation(parent: _listAnimCtrl, curve: Curves.easeOut);
    _listAnimCtrl.forward();
  }

  @override
  void dispose() {
    _searchCtrl.dispose();
    _scrollController.dispose();
    _listAnimCtrl.dispose();
    super.dispose();
  }

  Future<void> _loadFavorites() async {
    final prefs = await SharedPreferences.getInstance();
    final favList = prefs.getStringList('favorites') ?? [];
    setState(() {
      _favorites.addAll(favList);
      _favoritesLoaded = true;
    });
  }

  Future<void> _saveFavorites() async {
    final prefs = await SharedPreferences.getInstance();
    await prefs.setStringList('favorites', _favorites.toList());
  }

  void _toggleFavorite(String num) {
    setState(() {
      if (_favorites.contains(num)) {
        _favorites.remove(num);
      } else {
        _favorites.add(num);
      }
    });
    _saveFavorites();
  }

  String t(String en, String tr) => widget.settings.lang == 'tr' ? tr : en;

  void _copy(String text) {
    Clipboard.setData(ClipboardData(text: text));
    ScaffoldMessenger.of(context).showSnackBar(
      SnackBar(
        content: Row(
          children: [
            const Icon(Icons.check_circle_outline, color: Colors.white, size: 18),
            const SizedBox(width: 8),
            Text(t('Copied!', 'Kopyalandı!')),
          ],
        ),
        behavior: SnackBarBehavior.floating,
        shape: RoundedRectangleBorder(borderRadius: BorderRadius.circular(10)),
        duration: const Duration(milliseconds: 1200),
        margin: const EdgeInsets.all(12),
      ),
    );
  }

  void _showInfo(Map<String, dynamic> ch) {
    showGeneralDialog(
      context: context,
      barrierDismissible: true,
      barrierLabel: 'Info',
      barrierColor: Colors.black54,
      transitionDuration: const Duration(milliseconds: 350),
      transitionBuilder: (context, anim, secondaryAnim, child) {
        final curved = CurvedAnimation(parent: anim, curve: Curves.easeOutBack);
        return ScaleTransition(
          scale: curved,
          child: FadeTransition(opacity: anim, child: child),
        );
      },
      pageBuilder: (ctx, anim, secondaryAnim) => AlertDialog(
        shape: RoundedRectangleBorder(borderRadius: BorderRadius.circular(16)),
        title: Text('ATA ${ch['num']}'),
        content: SingleChildScrollView(
          child: Text(
            ch['info'][widget.settings.lang],
            style: const TextStyle(height: 1.5),
          ),
        ),
        actions: [
          TextButton(onPressed: () => Navigator.pop(ctx), child: Text(t('Close', 'Kapat'))),
        ],
      ),
    );
  }

  void _openSettings() {
    Navigator.push(
      context,
      PageRouteBuilder(
        pageBuilder: (context, animation, secondaryAnimation) => SettingsPage(
          settings: widget.settings,
          onSettingsChanged: widget.onSettingsChanged,
        ),
        transitionsBuilder: (context, animation, secondaryAnimation, child) {
          final curved = CurvedAnimation(parent: animation, curve: Curves.easeInOutCubic);
          return SlideTransition(
            position: Tween<Offset>(begin: const Offset(1.0, 0.0), end: Offset.zero).animate(curved),
            child: FadeTransition(opacity: curved, child: child),
          );
        },
        transitionDuration: const Duration(milliseconds: 400),
        reverseTransitionDuration: const Duration(milliseconds: 350),
      ),
    );
  }

  List<Map<String, dynamic>> get filteredData {
    final q = _searchCtrl.text.toLowerCase().trim();
    if (q.isEmpty) return ataData;

    return ataData.map((section) {
      final chapters = (section['chapters'] as List).where((ch) {
        final num = ch['num'] as String;
        final title = ch['title'][widget.settings.lang] as String;
        final subs = ch['subs'] as List<String>;
        return num.contains(q) || title.toLowerCase().contains(q) || subs.any((s) => s.toLowerCase().contains(q));
      }).toList();
      return {...section, 'chapters': chapters};
    }).where((s) => (s['chapters'] as List).isNotEmpty).toList();
  }

                                    }),
                                  ),
                                  AnimatedSize(
                                    duration: const Duration(milliseconds: 300),
                                    curve: Curves.easeInOut,
                                    child: isExpanded
                                        ? Column(
                                            children: subs.asMap().entries.map((subEntry) {
                                              final subIdx = subEntry.key;
                                              final sub = subEntry.value;
                                              return TweenAnimationBuilder<double>(
                                                key: ValueKey('$num-sub-$subIdx'),
                                                tween: Tween(begin: 0.0, end: 1.0),
                                                duration: Duration(milliseconds: 200 + (subIdx * 40)),
                                                curve: Curves.easeOutCubic,
                                                builder: (context, value, child) {
                                                  return Opacity(
                                                    opacity: value,
                                                    child: Transform.translate(
                                                      offset: Offset(20 * (1 - value), 0),
                                                      child: child,
                                                    ),
                                                  );
                                                },
                                                child: ListTile(
                                                  dense: true,
                                                  contentPadding: const EdgeInsets.only(left: 24, right: 10),
                                                  title: Text(
                                                    sub,
                                                    style: TextStyle(
                                                      fontSize: widget.settings.fontSize - 1,
                                                      color: widget.settings.textColor.withOpacity(0.9),
                                                      fontFamily: widget.settings.fontFamily,
                                                    ),
                                                  ),
                                                  onTap: () => _copy(sub),
                                                ),
                                              );
                                            }).toList(),
                                          )
                                        : const SizedBox.shrink(),
                                  ),
                                ],
                              ),
                            ),
                          ),
                        );
                      }),
                    ],
                  );
                },
              ),
            ),
          ],
        ),
      ),
    );
  }
}

class SettingsPage extends StatefulWidget {
  final AppSettings settings;
  final ValueChanged<AppSettings> onSettingsChanged;

  const SettingsPage({super.key, required this.settings, required this.onSettingsChanged});

  @override
  State<SettingsPage> createState() => _SettingsPageState();
}

class _SettingsPageState extends State<SettingsPage> {
  late AppSettings _settings;

  @override
  void initState() {
    super.initState();
    _settings = widget.settings;
  }

  void _updateSetting(AppSettings newSettings) {
    setState(() => _settings = newSettings);
    widget.onSettingsChanged(newSettings);
  }

  String t(String en, String tr) => _settings.lang == 'tr' ? tr : en;

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text(t('Settings', 'Ayarlar')),
      ),
      body: ListView(
        padding: const EdgeInsets.all(16),
        children: [
          SwitchListTile(
            title: Text(t('Dark Mode', 'Karanlık Mod')),
            value: _settings.isDark,
            onChanged: (v) => _updateSetting(_settings.copyWith(isDark: v)),
          ),
          const Divider(),
          ListTile(
            title: Text(t('Font Family', 'Yazı Tipi')),
            subtitle: Text(_settings.fontFamily),
            trailing: DropdownButton<String>(
              value: _settings.fontFamily,
              items: ['Roboto', 'Arial', 'Courier'].map((f) => DropdownMenuItem(value: f, child: Text(f))).toList(),
              onChanged: (v) => v == null ? null : _updateSetting(_settings.copyWith(fontFamily: v)),
            ),
          ),
          ListTile(
            title: Text(t('Font Size', 'Yazı Boyutu')),
            subtitle: Slider(
              value: _settings.fontSize,
              min: 10,
              max: 20,
              divisions: 10,
              label: _settings.fontSize.toStringAsFixed(0),
              onChanged: (v) => _updateSetting(_settings.copyWith(fontSize: v)),
            ),
          ),
          ListTile(
            title: Text(t('Line Height', 'Satır Aralığı')),
            subtitle: Slider(
              value: _settings.lineHeight,
              min: 1.0,
              max: 2.0,
              divisions: 10,
              label: _settings.lineHeight.toStringAsFixed(1),
              onChanged: (v) => _updateSetting(_settings.copyWith(lineHeight: v)),
            ),
          ),
          const Divider(),
          ElevatedButton(
            onPressed: () {
              _settings.resetToDefault();
              _updateSetting(_settings);
            },
            child: Text(t('Reset to Default', 'Varsayılana Dön')),
          ),
        ],
      ),
    );
  }
}
